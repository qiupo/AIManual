```python
2. 去除重复
3. 格式统一

# AI生成代码
import pandas as pd

# 处理缺失值
df.fillna(method='ffill', inplace=True)

# 去除重复
df.drop_duplicates(inplace=True)

# 格式统一
df['date'] = pd.to_datetime(df['date'])

# 异常值检测
Q1 = df['age'].quantile(0.25)
Q3 = df['age'].quantile(0.75)
IQR = Q3 - Q1
df = df[~((df['age'] < (Q1 - 1.5*IQR)) |
         (df['age'] > (Q3 + 1.5*IQR)))]
```
